Установка

Вставьте код на ваш сайт в место, где должна располагаться кнопка "Оплатить":

<div id = "qr-container"></div>
<form name = "iceBitsyPayForm" onsubmit="iceDevApi.qrCodeSubmit(); return false;">
    <input class = "iceBitsyPayRow" type = "hidden" name = "shopAuthCode" value = "MyshopAuthCode">
    <input class = "iceBitsyPayRow" type = "hidden" name = "code" value = "MyCode">
    <input class = "iceBitsyPayRow" type = "hidden" name = "currency" value = "RUB">
    <input class = "iceBitsyPayRow" type = "text" name = "amount" placeholder = "Сумма заказа" value = "" required>
    <input class = "iceBitsyPayRow" type = "text" name = "order" placeholder = "Номер заказа" value = "">
    <input class = "iceBitsyPayRow" type = "text" name = "description" placeholder = "Описание заказа" value = "">
    <input class = "iceBitsyPayRow" type = "text" name = "name" placeholder = "ФИО плательщика" value = "">
    <input class = "iceBitsyPayRow" type = "text" name = "email" placeholder = "E-mail" value = "">
    <input class = "iceBitsyPayRow" type = "text" name = "phone" placeholder = "Контактный телефон" value = "">
    <input class = "iceBitsyPayRow" type = "submit" name = "payButton" value = "Оплатить">
<form/>
